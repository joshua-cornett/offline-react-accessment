/**
 * @fileoverview Provides a context for managing the state of the form components
 */

import { createContext, useContext, useState } from 'react';
import { isNameValid as validateNameApi } from '../../mock-api/apis'; // import mock API logic

/**
 * Context to manage Form state. Prevents prop drilling.
 * @type {React.Context}
 */
const FormContext = createContext();

/**
 * Custom hook for the form context.
 * Ensures form state is accessible across components.
 * @function
 * @returns {object} Context containing form state and functions.
 */
export const useFormContext = () => useContext(FormContext);

/**
 * Provides a context wrapper for form validation.
 * @function
 * @param {object} props - Props object.
 * @param {React.ReactNode} props.children - The child components.
 * @returns {JSX.Element} The context provider.
 */
export const FormProvider = ({ children }) => {
  const [name, setName] = useState('');
  const [isNameValid, setIsNameValid] = useState(null);
  const [validating, setValidating] = useState(false); // New state to track validation status

  /**
   * Validates the name input with a mock API call.
   * @function
   * @param {string} name - The name to validate.
   * @returns {Promise<void>} Resolves when validation state is updated.
   */
  const validateName = async (name) => {
    setValidating(true); // Start validating

    try {
      const result = await validateNameApi(name); // mock API call
      setIsNameValid(result);
      console.log(result);
    } catch (error) {
      console.error('Error validating name:', error); // error for failed "API call"
      setIsNameValid(false);
    } finally {
      setValidating(false); // End validating
    }
  };

  return (
    <FormContext.Provider
      value={{
        name,
        setName,
        isNameValid,
        setIsNameValid,
        validateName,
        validating,
        setValidating,
      }}
    >
      {children}
    </FormContext.Provider>
  );
};
